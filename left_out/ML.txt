# ⚙️ 5. INFERENCE LAYER

## 📌 `inference.py`

```python
def run_xgb(model, features):
    if model is None:
        return None
    return model.predict_proba(features)[0][1]  # probability of disruption


def run_iso(model, features):
    if model is None:
        return None
    score = model.decision_function(features)[0]
    return 1 - score  # convert anomaly → risk
```

# 🔗 6. FUSION LOGIC (DRI CORE)

## 📌 `fusion.py`

```python
def compute_dri(context_score, anomaly_score):
    # fallback-safe
    context_score = context_score if context_score is not None else 0.5
    anomaly_score = anomaly_score if anomaly_score is not None else 0.5

    dri = (0.7 * context_score + 0.3 * anomaly_score) * 100

    return int(min(max(dri, 0), 100))
```

# 🔍 7. EXPLAINABILITY (SHAP-LIKE)

## 📌 `explain.py`

```python
def explain_features(data):
    contributions = {
        "port_congestion": data.get("port_congestion", 0) * 30,
        "weather": data.get("weather", 0) * 25,
        "carrier_risk": (1 - data.get("carrier_reliability", 1)) * 20,
        "customs_delay": data.get("customs_delay", 0) * 15,
        "geo_risk": data.get("geo_risk", 0) * 10,
    }

    sorted_features = sorted(contributions.items(), key=lambda x: -x[1])

    return sorted_features[:3]
```

# 🧠 8. FULL PIPELINE (MAIN FUNCTION)

## 📌 `pipeline.py`

```python
from ml.loader import load_models
from ml.pipeline.feature_engineering import build_feature_vector
from ml.pipeline.inference import run_xgb, run_iso
from ml.pipeline.fusion import compute_dri
from ml.pipeline.explain import explain_features

models = load_models()

def compute_dri_pipeline(data):
    features = build_feature_vector(data)

    xgb_score = run_xgb(models["xgb"], features)
    iso_score = run_iso(models["iso"], features)

    dri = compute_dri(xgb_score, iso_score)

    explanation = explain_features(data)

    return {
        "dri": dri,
        "top_factors": explanation
    }
```
