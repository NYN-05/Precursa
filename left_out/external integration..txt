# 🌦️ 4. WEATHER INTEGRATION

## 📌 `weather.py`

```python
import random
import requests
from .base import BaseAdapter

class WeatherAdapter(BaseAdapter):
    def __init__(self, use_mock=True):
        self.use_mock = use_mock

    def fetch(self, lat, lon):
        if self.use_mock:
            return self._mock_weather()
        return self._real_weather(lat, lon)

    def _mock_weather(self):
        return {
            "weather_severity": round(random.uniform(0, 1), 2),
            "wind_speed": random.randint(5, 40),
            "storm": random.choice([0, 1])
        }

    def _real_weather(self, lat, lon):
        try:
            url = f"https://api.open-meteo.com/v1/forecast?latitude={lat}&longitude={lon}&current_weather=true"
            res = requests.get(url, timeout=2).json()

            return {
                "weather_severity": min(res["current_weather"]["windspeed"] / 50, 1),
                "wind_speed": res["current_weather"]["windspeed"],
                "storm": 1 if res["current_weather"]["weathercode"] > 60 else 0
            }
        except:
            return self._mock_weather()
```

# 🏭 7. INTEGRATION FACTORY (IMPORTANT)

## 📌 `factory.py`

```python
from .weather import WeatherAdapter
from .ais import AISAdapter
from .events import EventsAdapter

class IntegrationFactory:
    def __init__(self, use_mock=True):
        self.weather = WeatherAdapter(use_mock)
        self.ais = AISAdapter(use_mock)
        self.events = EventsAdapter(use_mock)

    def fetch_all(self, shipment):
        weather = self.weather.fetch(
            shipment["lat"], shipment["lon"]
        )

        ais = self.ais.fetch(shipment["id"])

        events = self.events.fetch(shipment["origin"])

        return {**weather, **ais, **events}
```
